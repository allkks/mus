# --- ВСПОМОГАТЕЛЬНЫЕ МЕТОДЫ --- def add_artist(self): name = ctk.CTkInputDialog(text="Имя артиста:", title="Новый артист").get_input() if name and name.strip(): repo.create_file(f"{ROOT_DIR}/{name}/singles/.keep", "init", "") repo.create_file(f"{ROOT_DIR}/{name}/albums/.keep", "init", "") self.load_artists() def add_album(self): name = ctk.CTkInputDialog(text="Название альбома:", title="Новый альбом").get_input() if name and name.strip(): repo.create_file(f"{ROOT_DIR}/{self.current_artist}/albums/{name}/.keep", "init", "") self.open_artist(self.current_artist) def upload_track(self, subpath): path = filedialog.askopenfilename(filetypes=[("Audio", "*.mp3")]) if path: name = os.path.basename(path) with open(path, "rb") as f: content = f.read() repo.create_file(f"{ROOT_DIR}/{self.current_artist}/{subpath}/{name}", "upd", content) if self.current_album: self.open_album(self.current_album) else: self.open_artist(self.current_artist) def upload_artist_photo(self): path = filedialog.askopenfilename(filetypes=[("Image", "*.jpg;*.png")]) if path: with open(path, "rb") as f: content = f.read() p = f"{ROOT_DIR}/{self.current_artist}/artist.jpg" try: old = repo.get_contents(p) repo.update_file(old.path, "upd", content, old.sha) except: repo.create_file(p, "add", content) messagebox.showinfo("OK", "Фото загружено") def upload_album_cover(self): path = filedialog.askopenfilename(filetypes=[("Image", "*.jpg;*.png")]) if path: with open(path, "rb") as f: content = f.read() p = f"{ROOT_DIR}/{self.current_artist}/albums/{self.current_album}/cover.jpg" try: old = repo.get_contents(p) repo.update_file(old.path, "upd", content, old.sha) except: repo.create_file(p, "add", content) messagebox.showinfo("OK", "Обложка загружена") def delete_artist(self): if messagebox.askyesno("Удаление", f"Удалить {self.current_artist}?"): files = repo.get_contents(f"{ROOT_DIR}/{self.current_artist}", recursive=True) for f in reversed(files): if f.type == "file": repo.delete_file(f.path, "del", f.sha) self.load_artists() def delete_file(self, file_obj): repo.delete_file(file_obj.path, "del", file_obj.sha) if self.current_album: self.open_album(self.current_album) else: self.open_artist(self.current_artist) # --- СИНХРОНИЗАЦИЯ (ПОД НОВЫЙ САЙТ) --- def sync_database(self): db = {"artists": {}, "tracks": []} # Artists теперь объект! try: dirs = repo.get_contents(ROOT_DIR) track_count = 0 for d in dirs: if d.type == "dir": artist_img = f"https://raw.githubusercontent.com/{REPO_NAME}/main/{ROOT_DIR}/{d.name}/artist.jpg" db["artists"][d.name] = { "name": d.name, "img": artist_img, "tracks": [], "plays": 0 } # Синглы try: ss = repo.get_contents(f"{ROOT_DIR}/{d.name}/singles") for s in ss: if s.name.endswith(".mp3"): t_obj = { "id": track_count, "title": s.name[:-4], "artist": d.name, "url": s.download_url, "img": artist_img, "plays": 0 } db["tracks"].append(t_obj) db["artists"][d.name]["tracks"].append(t_obj) track_count += 1 except: pass # Альбомы try: albs = repo.get_contents(f"{ROOT_DIR}/{d.name}/albums") for al in albs: if al.type == "dir": alb_img = f"https://raw.githubusercontent.com/{REPO_NAME}/main/{al.path}/cover.jpg" ts = repo.get_contents(al.path) for t in ts: if t.name.endswith(".mp3"): t_obj = { "id": track_count, "title": t.name[:-4], "artist": d.name, "url": t.download_url, "img": alb_img, "plays": 0 } db["tracks"].append(t_obj) db["artists"][d.name]["tracks"].append(t_obj) track_count += 1 except: pass content = json.dumps(db, indent=4, ensure_ascii=False) try: old = repo.get_contents("database.json") repo.update_file("database.json", "sync", content, old.sha) except: repo.create_file("database.json", "sync", content) messagebox.showinfo("Успех", "Сайт обновлен!") except Exception as e: messagebox.showerror("Ошибка", str(e)) if __name__ == "__main__": PremiumManager().mainloop()
